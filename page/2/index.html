\begin{aligned}
\dot{U}{1}=Z{11}\dot{I}{1}+Z{12}\dot{I}{2}
\ \dot{U}{2}=Z_{21}\dot{I}{1}+Z{22}\dot{I}{2}
\end{aligned}
\right. U=\begin{bmatrix}
\dot{U}{1}\dot{U}{2}
\end{bmatrix}\qquad I=
\begin{bmatrix}
\dot{I}{1}\dot{I}{2}
\end{bmatrix} \qquad Z=\begin{bmatrix}
Z{11};Z_{12}\Z_{21};Z_{22}
\end{bmatrix} U=ZIZ_{12}=Z_{21}$$

\begin{aligned}
\dot{I}{1}=Y{11}\dot{U}{1}+Y{12}\dot{U}{2}

\end{aligned}
\right. Y=\begin{bmatrix}
Y{11};Y_{12}\Y_{21};Y_{22}
\end{bmatrix} I=YU $$
